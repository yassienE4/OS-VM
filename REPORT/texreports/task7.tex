How to run

\begin{itemize}
\item
  First move to working directory
\item
  cd chat
\item
  Compile both server and client
\item
  g++ server.cpp -o server -pthread
\item
  g++ client.cpp -o client -pthread
\item
  ./server
\item
  Then instantiate as many clients as needed
\item
  ./client
\end{itemize}
